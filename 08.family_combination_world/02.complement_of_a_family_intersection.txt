apply Subset.antisymm

intro x hx
rewrite [mem_compl_iff] at hx
rewrite [mem_sInter] at hx
push_neg at hx
obtain ⟨t, ⟨htF, hxt⟩⟩ := hx
rewrite [← mem_compl_iff] at hxt
rewrite [← compl_compl t] at htF
use tᶜ
rewrite [mem_setOf]
exact And.intro htF hxt

intro x hx
obtain ⟨s, hs⟩ := hx
rewrite [mem_setOf] at hs
intro hxF
have hxsc := hxF sᶜ hs.left
exact hxsc hs.right

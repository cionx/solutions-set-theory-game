apply Subset.antisymm

-- A ∩ (B ∪ C) ⊆ (A ∩ B) ∪ (A ∩ C)
intro x habc
have ha : x ∈ A := habc.left
have hbc : x ∈ B ∪ C := habc.right
rcases hbc with hb | hc
exact Or.inl (And.intro ha hb)
exact Or.inr (And.intro ha hc)

-- (A ∩ B) ∪ (A ∩ C) ⊆ A ∩ (B ∪ C)
intro x habac
rcases habac with hab | hac
have ha : x ∈ A := hab.left
have hb : x ∈ B := hab.right
exact And.intro ha (Or.inl hb)
have ha : x ∈ A := hac.left
have hc : x ∈ C := hac.right
exact And.intro ha (Or.inr hc)

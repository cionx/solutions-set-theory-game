-- A ∩ B  →  ⋂₀ {A, B}
intro hx
intro t ht
rcases ht with hA | hB
rewrite [hA]
exact hx.left
rewrite [hB]
exact hx.right

-- ⋂₀ {A, B} → A ∩ B
intro hx
have hA : A ∈ {A, B} -- One-line proof?
left
apply rfl
have hB : B ∈ {A, B} -- One-line proof?
right
rfl
constructor
exact hx A hA
exact hx B hB

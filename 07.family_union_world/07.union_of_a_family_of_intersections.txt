apply Subset.antisymm

-- A ∩ ⋃₀ F ⊆ ⋃₀ {s | ∃ u ∈ F, s = A ∩ u}
intro x ⟨hxA, hxF⟩
obtain ⟨u, ⟨huF, hxu⟩⟩ := hxF
rewrite [mem_sUnion]
apply Exists.intro (A ∩ u)
constructor
use u
exact And.intro hxA hxu

-- ⋃₀ {s | ∃ u ∈ F, s = A ∩ u} ⊆ A ∩ ⋃₀ F
intro x hx
obtain ⟨s, ⟨hs, hx⟩⟩ := hx
obtain ⟨u, ⟨huF, hsAu⟩⟩ := hs
rewrite [hsAu] at hx
have hxA : x ∈ A := hx.left
have hxu : x ∈ u := hx.right
constructor
exact hxA
use u

apply Subset.antisymm

-- A ∪ B ⊆ ⋃₀ {A, B}
intro x hxAB
rcases hxAB with hxA | hxB
-- Case x ∈ A.
have hA : A ∈ {A, B}
left
rfl
use A
-- Case x ∈ B.
have hB : B ∈ {A, B}
right
rfl
use B

-- ⋃₀ {A, B} ⊆ A ∪ B
intro x hxAB
obtain ⟨s, ⟨hxAB2, hxs⟩⟩ := hxAB
rcases hxAB2 with hxA | hxB
-- Case s = A.
left
rewrite [← hxA]
exact hxs
-- Case s = B.
right
rewrite [← hxB]
exact hxs
